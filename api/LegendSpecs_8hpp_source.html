468 inline auto LegendSpecs::repr() const -> std::string
469 {
470  const auto show = ShowSpecsOf<LegendSpecs>::repr();
471  if(show == "no")
472  return "unset key";
473 
474  auto titlespecs = TitleSpecsOf<LegendSpecs>::repr();
475  if(titlespecs.size())
476  titlespecs += " " + m_title_loc; // attach left|center|right to title (e.g. title 'Legend' right)
477 
478  std::stringstream ss;
479  ss << "set key ";
480  ss << m_placement << " " << m_opaque << " ";
481  ss << m_alignment << " ";
482  ss << m_justification << " ";
483  ss << m_invert << " ";
484  ss << m_reverse << " ";
485  ss << "width " << m_width_increment << " ";
486  ss << "height " << m_height_increment << " ";
487  ss << "samplen " << m_samplen << " ";
488  ss << "spacing " << m_spacing << " ";
489  ss << TextSpecsOf<LegendSpecs>::repr() << " ";
490  ss << titlespecs << " ";
491  ss << FrameSpecsOf<LegendSpecs>::repr() << " ";
492  ss << "maxrows " << m_maxrows << " ";
493  ss << "maxcols " << m_maxcols << " ";
494  return internal::removeExtraWhitespaces(ss.str());
495 }
