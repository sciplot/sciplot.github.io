200 template <typename DerivedSpecs>
201 auto TicsSpecsBaseOf<DerivedSpecs>::repr(std::string axis) const -> std::string
202 {
203  const auto show = ShowSpecsOf<DerivedSpecs>::repr();
204  if(show == "no")
205  return "unset " + axis + "tics";
206 
207  std::stringstream ss;
208  ss << "set " + axis + "tics" << " ";
209  ss << m_along << " ";
210  ss << m_mirror << " ";
211  ss << m_inout << " ";
212  ss << "scale " << m_scalemajor << "," << m_scaleminor << " ";
213  ss << m_rotate << " ";
214  ss << OffsetSpecsOf<DerivedSpecs>::repr() << " ";
215  ss << TextSpecsOf<DerivedSpecs>::repr() << " ";
216  ss << m_format;
217  return internal::removeExtraWhitespaces(ss.str());
218 }
