139 inline auto DrawSpecs::repr() const -> std::string
140 {
141  std::string use = m_using;
142  if(m_xtic.size()) use += ":" + m_xtic;
143  if(m_ytic.size()) use += ":" + m_ytic;
144 
145  std::stringstream ss;
146  ss << m_what << " ";
147  ss << gnuplot::optionValueStr("using", use);
148  ss << m_title << " ";
149  ss << gnuplot::optionValueStr("with", m_with);
150  ss << LineSpecsOf<DrawSpecs>::repr() << " ";
151  ss << PointSpecsOf<DrawSpecs>::repr() << " ";
152  ss << FillSpecsOf<DrawSpecs>::repr() << " ";
153  return internal::removeExtraWhitespaces(ss.str());
154 }
