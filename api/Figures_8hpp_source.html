117 inline Figures::Figures(const std::initializer_list<std::initializer_list<Figure>>& figs)
118 : m_id(m_counter++),
119  m_scriptfilename("multishow" + internal::str(m_id) + ".plt")
120 {
121  m_layoutrows = figs.size();
122  m_layoutcols = 1;
123  for(const auto& row : figs)

172 inline auto Figures::show() const -> void
173 {
174  // Open script file and truncate it
175  std::ofstream script(m_scriptfilename);
176 
177  // Add palette info. Use default palette if the user hasn't set one
178  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
179 
180  // Add terminal info
181  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
182  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
183  std::string size = gnuplot::sizestr(width, height, false);
184  gnuplot::showterminalcmd(script, size);
185 
186  // Add multiplot commands
187  gnuplot::multiplotcmd(script, m_layoutrows, m_layoutcols, m_title);
188 
189  // // Add the plot commands
190  // for(const auto& p : m_figures)
191  // {
192  // script << p.repr();
193  // }
194 
195  // Add the plot commands
196  for(const auto& row : m_figures)
197  for(const auto& fig : row)
198  script << fig.repr();
199 
200  // Add an empty line at the end and close the script to avoid crashes with gnuplot
201  script << std::endl;
202  script.close();
203 
204  // save plot data to file(s)
205  saveplotdata();
206 
207  // Show the figure
208  gnuplot::runscript(m_scriptfilename, true);
209 
210  // remove the temporary files if user wants to
211  if(m_autoclean)
212  {
213  cleanup();
214  }
215 }

217 inline auto Figures::save(const std::string& filename) const -> void
218 {
219  // Clean the file name to prevent errors
220  auto cleanedfilename = gnuplot::cleanpath(filename);
221 
222  // Get extension from file name
223  auto extension = cleanedfilename.substr(cleanedfilename.rfind(".") + 1);
224 
225  // Open script file
226  std::ofstream script(m_scriptfilename);
227 
228  // Add palette info. Use default palette if the user hasn't set one
229  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
230 
231  // Add terminal info
232  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
233  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
234  std::string size = gnuplot::sizestr(width, height, extension == "pdf");
235  gnuplot::saveterminalcmd(script, extension, size);
236 
237  // Add output command
238  gnuplot::outputcmd(script, cleanedfilename);
239 
240  // Add multiplot commands
241  gnuplot::multiplotcmd(script, m_layoutrows, m_layoutcols, m_title);
242 
243  // // Add the plot commands
244  // for(const auto& p : m_figures)
245  // {
246  // script << p.repr();
247  // }
248 
249  // Add the plot commands
250  for(const auto& row : m_figures)
251  for(const auto& fig : row)
252  script << fig.repr();
253 
254  // Close multiplot
255  script << "unset multiplot" << std::endl;
256 
257  // Unset the output
258  script << std::endl;
259  script << "set output";
260 
261  // Add an empty line at the end and close the script to avoid crashes with gnuplot
262  script << std::endl;
263  script.close();
264 
265  // save plot data to file(s)
266  saveplotdata();
267 
268  // Save the figure as a file
269  gnuplot::runscript(m_scriptfilename, false);
270 
271  // remove the temporary files if user wants to
272  if(m_autoclean)
273  {
274  cleanup();
275  }
276 }
