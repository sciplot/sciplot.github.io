465 template <typename X, typename... Vecs>
466 inline auto Plot::drawWithVecs(const std::string& with, const X& x, const Vecs&... vecs) -> DrawSpecs&
467 {
468  // Write the given vectors x and y as a new data set to the stream
469  std::ostringstream datastream;
470  gnuplot::writedataset(datastream, m_numdatasets, x, vecs...);
471 
472  // Set the using string to "" if X is not vector of strings.
473  // Otherwise, x contain xtics strings. Set the `using` string
474  // so that these are properly used as xtics.
475  std::string use;
476  if constexpr (internal::isStringVector<X>)
477  {
478  const auto nvecs = sizeof...(Vecs);
479  use = "0:"; // here, column 0 means the pseudo column with numbers 0, 1, 2, 3...
480  for (auto i = 2; i <= nvecs + 1; ++i)
481  use += std::to_string(i) + ":"; // this constructs 0:2:3:4:
482  use += "xtic(1)"; // this terminates the string with 0:2:3:4:xtic(1), and thus column 1 is used for the xtics
483  }
484 
485  // Append new data set to existing data
486  m_data += datastream.str();
487 
488  // Draw the data saved using a data set with index `m_numdatasets`. Increase number of data sets and set the line style specification (desired behavior is 1, 2, 3 (incrementing as new lines are plotted)).
489  return draw("'" + m_datafilename + "' index " + internal::str(m_numdatasets++), use, with).lineStyle(m_drawspecs.size());
490 }
491 
492 template <typename X, typename... Vecs>
493 inline auto Plot::drawWithVecsContainingNaN(std::string with, const X& x, const Vecs&... vecs) -> DrawSpecs&
494 {
495  // Write the given vectors x and y as a new data set to the stream
496  std::ostringstream datastream;
497  gnuplot::writedataset(datastream, m_numdatasets, x, vecs...);
498 
499  std::string use;
500  const auto nvecs = sizeof...(Vecs);
501  use = "0:"; // here, column 0 means the pseudo column with numbers 0, 1, 2, 3...
502  for (auto i = 2; i <= nvecs + 1; ++i)
503  use += "($" + std::to_string(i) + "):"; // this constructs 0:$(2):$(3):$(4):
504  use += "xtic(1)"; // this terminates the string with 0:$(2):$(3):$(4):xtic(1), and thus column 1 is used for the xtics
505 
506  // Append new data set to existing data
507  m_data += datastream.str();
508 
509  // Draw the data saved using a data set with index `m_numdatasets`. Increase number of data sets and set the line style specification (desired behavior is 1, 2, 3 (incrementing as new lines are plotted)).
510  return draw("'" + m_datafilename + "' index " + internal::str(m_numdatasets++), use, with).lineStyle(m_drawspecs.size());
511 }

703 inline auto Plot::drawWithCols(const std::string& fname, const std::string& with, const std::vector<ColumnIndex>& cols) -> DrawSpecs&
704 {
705  std::string use;
706  for (const auto& col : cols)
707  use += col.value + ":"; // e.g., "1:4:5:7:" (where 1 is x, 4 is y, 5 is ylow and 7 is yhigh for a yerrorlines plot)
708  use = internal::trimright(use, ':'); // e.g., "1:4:5:7:" => "1:4:5:7"
709  std::string what = "'" + fname + "'"; // e.g., "'myfile.dat'"
710  return draw(what, use, with).lineStyle(m_drawspecs.size()); // e.g., draw(what="'myfile.dat'", use="1:2", with="lines");
711 }

852 inline auto Plot::repr() const -> std::string
853 {
854  std::stringstream script;
855 
856  // Add plot setup commands
857  script << "#==============================================================================" << std::endl;
858  script << "# SETUP COMMANDS" << std::endl;
859  script << "#==============================================================================" << std::endl;
860  script << gnuplot::commandValueStr("set xrange", m_xrange);
861  script << gnuplot::commandValueStr("set yrange", m_yrange);
862  script << m_xlabel << std::endl;
863  script << m_ylabel << std::endl;
864  script << m_zlabel << std::endl;
865  script << m_rlabel << std::endl;
866  script << m_border << std::endl;
867  script << m_grid << std::endl;
868  script << m_style_fill << std::endl;
869  script << m_style_histogram << std::endl;
870  script << m_tics << std::endl;
871  script << m_xtics_major_bottom << std::endl;
872  script << m_xtics_major_top << std::endl;
873  script << m_xtics_minor_bottom << std::endl;
874  script << m_xtics_minor_top << std::endl;
875  script << m_ytics_major_left << std::endl;
876  script << m_ytics_major_right << std::endl;
877  script << m_ytics_minor_left << std::endl;
878  script << m_ytics_minor_right << std::endl;
879  script << m_ztics_major << std::endl;
880  script << m_ztics_minor << std::endl;
881  script << m_rtics_major << std::endl;
882  script << m_rtics_minor << std::endl;
883  script << m_legend << std::endl;
884  script << gnuplot::commandValueStr("set boxwidth", m_boxwidth);
885  script << gnuplot::commandValueStr("set samples", m_samples);
886  script << gnuplot::commandValueStr("set datafile missing", MISSING_INDICATOR);
887 
888  // Add custom gnuplot commands
889  if (!m_customcmds.empty())
890  {
891  script << "#==============================================================================" << std::endl;
892  script << "# CUSTOM EXPLICIT GNUPLOT COMMANDS" << std::endl;
893  script << "#==============================================================================" << std::endl;
894  for (const auto& c : m_customcmds)
895  {
896  script << c << std::endl;
897  }
898  }
899 
900  // Add the actual plot commands for all drawXYZ() calls
901  script << "#==============================================================================" << std::endl;
902  script << "# PLOT COMMANDS" << std::endl;
903  script << "#==============================================================================" << std::endl;
904  script << "plot \\\n"; // use `\` to have a plot command in each individual line!
905 
906  // Write plot commands and style per plot
907  const auto n = m_drawspecs.size();
908  for (std::size_t i = 0; i < n; ++i)
909  script << " " << m_drawspecs[i] << (i < n - 1 ? ", \\\n" : ""); // consider indentation with 4 spaces!
910 
911  // Add an empty line at the end
912  script << std::endl;
913  return script.str();
914 }
