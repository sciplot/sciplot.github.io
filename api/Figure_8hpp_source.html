131 inline Figure::Figure(const std::initializer_list<std::initializer_list<PlotXD>>& plots)

204 inline auto Figure::show() const -> void
205 {
206  // Open script file and truncate it
207  std::ofstream script(m_scriptfilename);
208 
209  // Add palette info. Use default palette if the user hasn't set one
210  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
211 
212  // Add terminal info
213  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
214  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
215  std::string size = gnuplot::sizestr(width, height, false);
216  gnuplot::showterminalcmd(script, size, m_font);
217 
218  // Add multiplot commands
219  gnuplot::multiplotcmd(script, m_layoutrows, m_layoutcols, m_title);
220 
221  // Add the plot commands
222  for (const auto& row : m_plots)
223  {
224  for (const auto& plot : row)
225  {
226  if (auto* plot_p = std::get_if<Plot>(&plot))
227  {
228  script << plot_p->repr();
229  }
230  else if (auto* plot_p = std::get_if<Plot3D>(&plot))
231  {
232  script << plot_p->repr();
233  }
234  }
235  }
236 
237  // Add an empty line at the end and close the script to avoid crashes with gnuplot
238  script << std::endl;
239  script.close();
240 
241  // save plot data to file(s)
242  saveplotdata();
243 
244  // Show the figure
245  gnuplot::runscript(m_scriptfilename, true);
246 
247  // remove the temporary files if user wants to
248  if (m_autoclean)
249  {
250  cleanup();
251  }
252 }

254 inline auto Figure::save(const std::string& filename) const -> void
255 {
256  // Clean the file name to prevent errors
257  auto cleanedfilename = gnuplot::cleanpath(filename);
258 
259  // Get extension from file name
260  auto extension = cleanedfilename.substr(cleanedfilename.rfind(".") + 1);
261 
262  // Open script file
263  std::ofstream script(m_scriptfilename);
264 
265  // Add palette info. Use default palette if the user hasn't set one
266  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
267 
268  // Add terminal info
269  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
270  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
271  std::string size = gnuplot::sizestr(width, height, extension == "pdf");
272  gnuplot::saveterminalcmd(script, extension, size, m_font);
273 
274  // Add output command
275  gnuplot::outputcmd(script, cleanedfilename);
276 
277  // Add multiplot commands
278  gnuplot::multiplotcmd(script, m_layoutrows, m_layoutcols, m_title);
279 
280  // Add the plot commands
281  for (const auto& row : m_plots)
282  {
283  for (const auto& plot : row)
284  {
285  if (auto* plot_p = std::get_if<Plot>(&plot))
286  {
287  script << plot_p->repr();
288  }
289  else if (auto* plot_p = std::get_if<Plot3D>(&plot))
290  {
291  script << plot_p->repr();
292  }
293  }
294  }
295 
296  // Close multiplot
297  script << "unset multiplot" << std::endl;
298 
299  // Unset the output
300  script << std::endl;
301  script << "set output";
302 
303  // Add an empty line at the end and close the script to avoid crashes with gnuplot
304  script << std::endl;
305  script.close();
306 
307  // save plot data to file(s)
308  saveplotdata();
309 
310  // Save the figure as a file
311  gnuplot::runscript(m_scriptfilename, false);
312 
313  // remove the temporary files if user wants to
314  if (m_autoclean)
315  {
316  cleanup();
317  }
318 }
