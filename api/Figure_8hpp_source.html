543 inline auto Figure::draw(std::string what, std::string use, std::string with) -> DrawSpecs&
544 {
545  // Save the draw arguments for this x,y data
546  m_drawspecs.emplace_back(what, use, with);
547 
548  // Set the default line style specification for this drawing (desired behavior is 1, 2, 3 (incrementing as new lines are plotted))
549  m_drawspecs.back().lineStyle(m_drawspecs.size());
550 
551  // Return the just created drawing object in case the user wants to customize it
552  return m_drawspecs.back();
553 }

555 template <typename X, typename... Vecs>
556 inline auto Figure::drawWithVecs(std::string with, const X& x, const Vecs&... vecs) -> DrawSpecs&
557 {
558  // Write the given vectors x and y as a new data set to the stream
559  std::ostringstream datastream;
560  gnuplot::writedataset(datastream, m_numdatasets, x, vecs...);
561 
562  // Set the using string to "" if X is not vector of strings.
563  // Otherwise, x contain xtics strings. Set the `using` string
564  // so that these are properly used as xtics.
565  std::string use;
566  if constexpr(internal::isStringVector<X>) {
567  const auto nvecs = sizeof...(Vecs);
568  use = "0:"; // here, column 0 means the pseudo column with numbers 0, 1, 2, 3...
569  for(auto i = 2; i <= nvecs + 1; ++i)
570  use += std::to_string(i) + ":"; // this constructs 0:2:3:4:
571  use += "xtic(1)"; // this terminates the string with 0:2:3:4:xtic(1), and thus column 1 is used for the xtics
572  }
573 
574  // Append new data set to existing data
575  m_data += datastream.str();
576 
577  // Draw the data saved using a data set with index `m_numdatasets`. Increase number of data sets
578  return draw("'" + m_datafilename + "' index " + internal::str(m_numdatasets++), use, with);
579 }

747 inline auto Figure::drawWithCols(std::string fname, std::string with, const std::vector<ColumnIndex>& cols) -> DrawSpecs&

753  std::string what = "'" + fname + "'"; // e.g., "'myfile.dat'"
754  return draw(what, use, with); // e.g., draw(what="'myfile.dat'", use="1:2", with="lines");
755 }

911 inline auto Figure::show() const -> void
912 {
913  // Open script file and truncate it
914  std::ofstream script(m_scriptfilename);
915 
916  // Add palette info. Use default palette if the user hasn't set one
917  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
918 
919  // Add terminal info
920  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
921  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
922  std::string size = gnuplot::sizestr(width, height, false);
923  gnuplot::showterminalcmd(script, size);
924 
925  // Add the plot commands
926  script << repr();
927 
928  // Add an empty line at the end and close the script to avoid crashes with gnuplot
929  script << std::endl;
930  script.close();
931 
932  // save plot data to a file
933  savePlotData();
934 
935  // Show the figure
936  gnuplot::runscript(m_scriptfilename, true);
937 
938  // remove the temporary files if user wants to
939  if(m_autoclean)
940  {
941  cleanup();
942  }
943 }

945 inline auto Figure::save(std::string filename) const -> void
946 {
947  // Clean the file name to prevent errors
948  auto cleanedfilename = gnuplot::cleanpath(filename);
949 
950  // Get extension from file name
951  auto extension = cleanedfilename.substr(cleanedfilename.rfind(".") + 1);
952 
953  // Open script file
954  std::ofstream script(m_scriptfilename);
955 
956  // Add palette info. Use default palette if the user hasn't set one
957  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
958 
959  // Add terminal info
960  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
961  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
962  std::string size = gnuplot::sizestr(width, height, extension == "pdf");
963  gnuplot::saveterminalcmd(script, extension, size);
964 
965  // Add output command
966  gnuplot::outputcmd(script, cleanedfilename);
967 
968  // Add the plot commands
969  script << repr();
970 
971  // Unset the output
972  script << std::endl;
973  script << "set output";
974 
975  // Add an empty line at the end and close the script to avoid crashes with gnuplot
976  script << std::endl;
977  script.close();
978 
979  // save plot data to a file
980  savePlotData();
981 
982  // Save the figure as a file
983  gnuplot::runscript(m_scriptfilename, false);
984 
985  // remove the temporary files if user wants to
986  if (m_autoclean)
987  {
988  cleanup();
989  }
990 }
