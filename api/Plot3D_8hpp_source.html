335 template <typename X, typename... Vecs>
336 inline auto Plot3D::drawWithVecs(const std::string&with, const X& x, const Vecs&... vecs) -> DrawSpecs&
337 {
338  // Write the given vectors x and y as a new data set to the stream
339  std::ostringstream datastream;
340  gnuplot::writedataset(datastream, m_numdatasets, x, vecs...);
341 
342  // Set the using string to "" if X is not vector of strings.
343  // Otherwise, x contain xtics strings. Set the `using` string
344  // so that these are properly used as xtics.
345  std::string use;
346  if constexpr(internal::isStringVector<X>) {
347  const auto nvecs = sizeof...(Vecs);
348  use = "0:"; // here, column 0 means the pseudo column with numbers 0, 1, 2, 3...
349  for(auto i = 2; i <= nvecs + 1; ++i)
350  use += std::to_string(i) + ":"; // this constructs 0:2:3:4:
351  use += "xtic(1)"; // this terminates the string with 0:2:3:4:xtic(1), and thus column 1 is used for the xtics
352  }
353 
354  // Append new data set to existing data
355  m_data += datastream.str();
356 
357  // Draw the data saved using a data set with index `m_numdatasets`. Increase number of data sets and set the line style specification (desired behavior is 1, 2, 3 (incrementing as new lines are plotted)).
358  return draw("'" + m_datafilename + "' index " + internal::str(m_numdatasets++), use, with).lineStyle(m_drawspecs.size());;
359 }

402 inline auto Plot3D::drawWithCols(const std::string &fname, const std::string &with, const std::vector<ColumnIndex>& cols) -> DrawSpecs&
403 {
404  std::string use;
405  for(auto col : cols)
406  use += col.value + ":"; // e.g., "1:4:5:7:" (where 1 is x, 4 is y, 5 is ylow and 7 is yhigh for a yerrorlines plot)
407  use = internal::trimright(use, ':'); // e.g., "1:4:5:7:" => "1:4:5:7"
408  std::string what = "'" + fname + "'"; // e.g., "'myfile.dat'"
409  return draw(what, use, with).lineStyle(m_drawspecs.size());; // e.g., draw(what="'myfile.dat'", use="1:2", with="lines");
410 }

446 inline auto Plot3D::repr() const -> std::string
447 {
448  std::stringstream script;
449 
450  // Add plot setup commands
451  script << "#==============================================================================" << std::endl;
452  script << "# SETUP COMMANDS" << std::endl;
453  script << "#==============================================================================" << std::endl;
454  script << gnuplot::commandValueStr("set xrange", m_xrange);
455  script << gnuplot::commandValueStr("set yrange", m_yrange);
456  script << gnuplot::commandValueStr("set zrange", m_zrange);
457  script << m_xlabel << std::endl;
458  script << m_ylabel << std::endl;
459  script << m_zlabel << std::endl;
460  script << m_rlabel << std::endl;
461  script << m_border << std::endl;
462  script << m_grid << std::endl;
463  script << m_style_fill << std::endl;
464  script << m_style_histogram << std::endl;
465  script << m_tics << std::endl;
466  script << m_xtics_major_bottom << std::endl;
467  script << m_xtics_major_top << std::endl;
468  script << m_xtics_minor_bottom << std::endl;
469  script << m_xtics_minor_top << std::endl;
470  script << m_ytics_major_left << std::endl;
471  script << m_ytics_major_right << std::endl;
472  script << m_ytics_minor_left << std::endl;
473  script << m_ytics_minor_right << std::endl;
474  script << m_ztics_major << std::endl;
475  script << m_ztics_minor << std::endl;
476  script << m_rtics_major << std::endl;
477  script << m_rtics_minor << std::endl;
478  script << m_legend << std::endl;
479  script << gnuplot::commandValueStr("set boxwidth", m_boxwidth);
480  script << gnuplot::commandValueStr("set samples", m_samples);
481 
482  // Add custom gnuplot commands
483  if (!m_customcmds.empty())
484  {
485  script << "#==============================================================================" << std::endl;
486  script << "# CUSTOM EXPLICIT GNUPLOT COMMANDS" << std::endl;
487  script << "#==============================================================================" << std::endl;
488  for(const auto& c : m_customcmds)
489  {
490  script << c << std::endl;
491  }
492  }
493 
494  // Add the actual plot commands for all drawXYZ() calls
495  script << "#==============================================================================" << std::endl;
496  script << "# PLOT COMMANDS" << std::endl;
497  script << "#==============================================================================" << std::endl;
498  script << "splot \\\n"; // use `\` to have a plot command in each individual line!
499 
500  // Write plot commands and style per plot
501  const auto n = m_drawspecs.size();
502  for(std::size_t i = 0; i < n; ++i)
503  script << " " << m_drawspecs[i] << (i < n - 1 ? ", \\\n" : ""); // consider indentation with 4 spaces!
504 
505  // Add an empty line at the end
506  script << std::endl;
507  return script.str();
508 }
