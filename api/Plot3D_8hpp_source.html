158 template <typename X, typename... Vecs>
159 inline auto Plot3D::drawWithVecs(const std::string& with, const X& x, const Vecs&... vecs) -> DrawSpecs&
160 {
161  // Write the given vectors x and y as a new data set to the stream
162  std::ostringstream datastream;
163  gnuplot::writedataset(datastream, m_numdatasets, x, vecs...);
164 
165  // Set the using string to "" if X is not vector of strings.
166  // Otherwise, x contain xtics strings. Set the `using` string
167  // so that these are properly used as xtics.
168  std::string use;
169  if constexpr (internal::isStringVector<X>)
170  {
171  const auto nvecs = sizeof...(Vecs);
172  use = "0:"; // here, column 0 means the pseudo column with numbers 0, 1, 2, 3...
173  for (auto i = 2; i <= nvecs + 1; ++i)
174  use += std::to_string(i) + ":"; // this constructs 0:2:3:4:
175  use += "xtic(1)"; // this terminates the string with 0:2:3:4:xtic(1), and thus column 1 is used for the xtics
176  }
177 
178  // Append new data set to existing data
179  m_data += datastream.str();
180 
181  // Draw the data saved using a data set with index `m_numdatasets`. Increase number of data sets and set the line style specification (desired behavior is 1, 2, 3 (incrementing as new lines are plotted)).
182  return draw("'" + m_datafilename + "' index " + internal::str(m_numdatasets++), use, with).lineStyle(static_cast<int> (m_drawspecs.size()));
183  ;
184 }

227 inline auto Plot3D::drawWithCols(const std::string& fname, const std::string& with, const std::vector<ColumnIndex>& cols) -> DrawSpecs&
228 {
229  std::string use;
230  for (auto col : cols)
231  use += col.value + ":"; // e.g., "1:4:5:7:" (where 1 is x, 4 is y, 5 is ylow and 7 is yhigh for a yerrorlines plot)
232  use = internal::trimright(use, ':'); // e.g., "1:4:5:7:" => "1:4:5:7"
233  std::string what = "'" + fname + "'"; // e.g., "'myfile.dat'"
234  return draw(what, use, with).lineStyle(static_cast<int> (m_drawspecs.size()));
235  ; // e.g., draw(what="'myfile.dat'", use="1:2", with="lines");
236 }

272 inline auto Plot3D::repr() const -> std::string
273 {
274  std::stringstream script;
275 
276  // Add plot setup commands
277  script << "#==============================================================================" << std::endl;
278  script << "# SETUP COMMANDS" << std::endl;
279  script << "#==============================================================================" << std::endl;
280  script << gnuplot::commandValueStr("set xrange", m_xrange);
281  script << gnuplot::commandValueStr("set yrange", m_yrange);
282  script << gnuplot::commandValueStr("set zrange", m_zrange);
283  script << m_xlabel << std::endl;
284  script << m_ylabel << std::endl;
285  script << m_zlabel << std::endl;
286  script << m_rlabel << std::endl;
287  script << m_border << std::endl;
288  script << m_grid << std::endl;
289  script << m_style_fill << std::endl;
290  script << m_style_histogram << std::endl;
291  script << m_tics << std::endl;
292  script << m_xtics_major_bottom << std::endl;
293  script << m_xtics_major_top << std::endl;
294  script << m_xtics_minor_bottom << std::endl;
295  script << m_xtics_minor_top << std::endl;
296  script << m_ytics_major_left << std::endl;
297  script << m_ytics_major_right << std::endl;
298  script << m_ytics_minor_left << std::endl;
299  script << m_ytics_minor_right << std::endl;
300  script << m_ztics_major << std::endl;
301  script << m_ztics_minor << std::endl;
302  script << m_rtics_major << std::endl;
303  script << m_rtics_minor << std::endl;
304  script << m_legend << std::endl;
305  script << gnuplot::commandValueStr("set boxwidth", m_boxwidth);
306  script << gnuplot::commandValueStr("set samples", m_samples);
307 
308  // Add custom gnuplot commands
309  if (!m_customcmds.empty())
310  {
311  script << "#==============================================================================" << std::endl;
312  script << "# CUSTOM EXPLICIT GNUPLOT COMMANDS" << std::endl;
313  script << "#==============================================================================" << std::endl;
314  for (const auto& c : m_customcmds)
315  {
316  script << c << std::endl;
317  }
318  }
319 
320  // Add the actual plot commands for all drawXYZ() calls
321  script << "#==============================================================================" << std::endl;
322  script << "# PLOT COMMANDS" << std::endl;
323  script << "#==============================================================================" << std::endl;
324  script << "splot \\\n"; // use `\` to have a plot command in each individual line!
325 
326  // Write plot commands and style per plot
327  const auto n = m_drawspecs.size();
328  for (std::size_t i = 0; i < n; ++i)
329  script << " " << m_drawspecs[i] << (i < n - 1 ? ", \\\n" : ""); // consider indentation with 4 spaces!
330 
331  // Add an empty line at the end
332  script << std::endl;
333  return script.str();
334 }
