248 {

251  out << "#==============================================================================" << std::endl;
252  // We set a terminal here to make sure we can also set a size. This is necessary, because the
253  // canvas size can ONLY be set using "set terminal <TERMINAL> size W, H".
254  // See: http://www.bersch.net/gnuplot-doc/canvas-size.html#set-term-size
255  // The GNUTERM variable contains the default terminal, which we're using for the show command.
256  // See: http://www.bersch.net/gnuplot-doc/unset.html

258  //out << "set termoption font '" << DEFAULT_FONTNAME << "," << DEFAULT_FONTSIZE << "'" << std::endl;
