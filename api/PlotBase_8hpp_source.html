261 inline PlotBase::PlotBase()
262  : m_id(m_counter++), m_scriptfilename("show" + internal::str(m_id) + ".plt"), m_datafilename("plot" + internal::str(m_id) + ".dat"), m_xlabel("x"), m_ylabel("y"), m_rlabel("r"), m_xtics_major_bottom("x"), m_xtics_major_top("x2"), m_xtics_minor_bottom("x"), m_xtics_minor_top("x2"), m_ytics_major_left("y"), m_ytics_major_right("y2"), m_ytics_minor_left("y"), m_ytics_minor_right("y2"), m_ztics_major("z"), m_ztics_minor("z"), m_rtics_major("r"), m_rtics_minor("r")
263 {
264  // Show only major and minor xtics and ytics
265  xticsMajorBottom().show();
266  xticsMinorBottom().show();
267  yticsMajorLeft().show();
268  yticsMinorLeft().show();
269 
270  // Hide all other tics
271  xticsMajorTop().hide();
272  xticsMinorTop().hide();
273  yticsMajorRight().hide();
274  yticsMinorRight().hide();
275  zticsMajor().hide();
276  zticsMinor().hide();

287  // This is needed because of how drawHistogram works. Using `with histograms` don't work as well.
288  gnuplot("set style data histogram");

378 inline auto PlotBase::show() const -> void
379 {
380  // Open script file and truncate it
381  std::ofstream script(m_scriptfilename);
382 
383  // Add palette info. Use default palette if the user hasn't set one
384  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
385 
386  // Add terminal info
387  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
388  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
389  std::string size = gnuplot::sizestr(width, height, false);
390  gnuplot::showterminalcmd(script, size, m_font);
391 
392  // Add the plot commands
393  script << repr();
394 
395  // Add an empty line at the end and close the script to avoid crashes with gnuplot
396  script << std::endl;
397  script.close();
398 
399  // save plot data to a file
400  savePlotData();
401 
402  // Show the plot
403  gnuplot::runscript(m_scriptfilename, true);
404 
405  // remove the temporary files if user wants to
406  if (m_autoclean)
407  {
408  cleanup();
409  }
410 }

412 inline auto PlotBase::save(const std::string& filename) const -> void
413 {
414  // Clean the file name to prevent errors
415  auto cleanedfilename = gnuplot::cleanpath(filename);
416 
417  // Get extension from file name
418  auto extension = cleanedfilename.substr(cleanedfilename.rfind('.') + 1);
419 
420  // Open script file
421  std::ofstream script(m_scriptfilename);
422 
423  // Add palette info. Use default palette if the user hasn't set one
424  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
425 
426  // Add terminal info
427  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
428  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
429  std::string size = gnuplot::sizestr(width, height, extension == "pdf");
430  gnuplot::saveterminalcmd(script, extension, size, m_font);
431 
432  // Add output command
433  gnuplot::outputcmd(script, cleanedfilename);
434 
435  // Add the plot commands
436  script << repr();
437 
438  // Unset the output
439  script << std::endl;
440  script << "set output";
441 
442  // Add an empty line at the end and close the script to avoid crashes with gnuplot
443  script << std::endl;
444  script.close();
445 
446  // save plot data to a file
447  savePlotData();
448 
449  // Save the plot as a file
450  gnuplot::runscript(m_scriptfilename, false);
451 
452  // remove the temporary files if user wants to
453  if (m_autoclean)
454  {
455  cleanup();
456  }
457 }
