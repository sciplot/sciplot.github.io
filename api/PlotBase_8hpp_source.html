231 inline auto PlotBase::show() const -> void
232 {
233  // Open script file and truncate it
234  std::ofstream script(m_scriptfilename);
235 
236  // Add palette info. Use default palette if the user hasn't set one
237  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
238 
239  // Add terminal info
240  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
241  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
242  std::string size = gnuplot::sizestr(width, height, false);
243  gnuplot::showterminalcmd(script, size, m_font);
244 
245  // Add the plot commands
246  script << repr();
247 
248  // Add an empty line at the end and close the script to avoid crashes with gnuplot
249  script << std::endl;
250  script.close();
251 
252  // save plot data to a file
253  savePlotData();
254 
255  // Show the plot
256  gnuplot::runscript(m_scriptfilename, true);
257 
258  // remove the temporary files if user wants to
259  if(m_autoclean)
260  {
261  cleanup();
262  }
263 }

265 inline auto PlotBase::save(const std::string &filename) const -> void
266 {
267  // Clean the file name to prevent errors
268  auto cleanedfilename = gnuplot::cleanpath(filename);
269 
270  // Get extension from file name
271  auto extension = cleanedfilename.substr(cleanedfilename.rfind('.') + 1);
272 
273  // Open script file
274  std::ofstream script(m_scriptfilename);
275 
276  // Add palette info. Use default palette if the user hasn't set one
277  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
278 
279  // Add terminal info
280  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
281  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
282  std::string size = gnuplot::sizestr(width, height, extension == "pdf");
283  gnuplot::saveterminalcmd(script, extension, size, m_font);
284 
285  // Add output command
286  gnuplot::outputcmd(script, cleanedfilename);
287 
288  // Add the plot commands
289  script << repr();
290 
291  // Unset the output
292  script << std::endl;
293  script << "set output";
294 
295  // Add an empty line at the end and close the script to avoid crashes with gnuplot
296  script << std::endl;
297  script.close();
298 
299  // save plot data to a file
300  savePlotData();
301 
302  // Save the plot as a file
303  gnuplot::runscript(m_scriptfilename, false);
304 
305  // remove the temporary files if user wants to
306  if (m_autoclean)
307  {
308  cleanup();
309  }
310 }
