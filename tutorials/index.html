    Plotting sine functions¶

    #include <sciplot/sciplot.hpp>
using namespace sciplot;

int main(int argc, char** argv)
{
    // Create a vector with values from 0 to pi divived into 200 uniform intervals for the x-axis
    Vec x = linspace(0.0, PI, 200);

    // Create a Figure object
    Figure fig;

    // Set the x and y labels
    fig.xlabel("x");
    fig.ylabel("y");

    // Set the x and y ranges
    fig.xrange(0.0, PI);
    fig.yrange(0.0, 6.0);

    // Plot i*sin(x) from i = 1 to i = 6
    for (auto i = 1; i <= 6; ++i)
        fig.drawCurve(x, i * std::sin(x)).label(std::to_string(i) + "⋅sin(x)");

    // Show the figure in a pop-up window
    fig.show();

    // Save the figure to a PDF file
    fig.save("example-sine-functions.pdf");
}

    Plotting multiple plots¶

    #include <sciplot/sciplot.hpp>
using namespace sciplot;

int main(int argc, char** argv)
{
    // Create a vector with values from 0 to 5 divived into 200 uniform intervals for the x-axis
    Vec x = linspace(0.0, 5.0, 200);

    // Create 4 different figures
    Figure fig0;
    fig0.drawCurve(x, std::sin(x)).label("sin(x)");
    fig0.drawCurve(x, std::cos(x)).label("cos(x)");

    Figure fig1;
    fig1.drawCurve(x, std::cos(x)).label("cos(x)");

    Figure fig2;
    fig2.drawCurve(x, std::tan(x)).label("tan(x)");

    Figure fig3;
    fig3.drawCurve(x, std::sqrt(x)).label("sqrt(x)");

    // Use the previous figures as sub-figures in a larger 2x2 figure.
    Figures fig = {{ fig0, fig1 },
                   { fig2, fig3 }};

    fig.title("Trigonometric Functions");
    fig.palette("dark2");

    // Save the figure to a PDF file
    fig.save("example-multiplot.pdf");
}

    Plotting boxes¶

    #include <sciplot/sciplot.hpp>
using namespace sciplot;

int main(int argc, char** argv)
{
    // Create a vector with x-axis values
    std::vector<int> x = {0, 1, 2, 3};

    // Create a vector with y values
    std::vector<float> y = {-4, 2, 5, -3};

    // Create a Figure object
    Figure fig;

    // Set the legend
    fig.legend().hide();

    // Set the x and y labels
    fig.xlabel("x");
    fig.ylabel("y");

    // Set the y range
    fig.yrange(-5, 5);

    // Add values to plot
    fig.drawBoxes(x, y)
        .fillSolid()
        .fillColor("green")
        .fillIntensity(0.5);

    // Adjust the relative width of the boxes
    fig.boxWidthRelative(0.75);

    // Show the figure in a pop-up window
    fig.show();

    // Save the figure to a PDF file
    fig.save("example-boxes.pdf");
}

    Plotting boxes with custom tick labels¶

    #include <sciplot/sciplot.hpp>
using namespace sciplot;

int main(int argc, char** argv)
{
    // Create a vector with the xtic labels for the boxes
    Strings names = {"John", "Peter", "Thomas", "Marta"};

    // Create a vector with the y values for the boxes
    Vec ages = {44, 27, 35, 20};

    // Create a vector with the xwidth values for the boxes
    Vec experiences = {0.8, 0.4, 0.7, 0.9};

    // Create a Figure object
    Figure fig;

    // Set the legend to the top left corner of the figure
    fig.legend().atTopLeft();

    // Set the y label and its range
    fig.ylabel("Age");
    fig.yrange(0.0, 50);

    // Plot the boxes using y values.
    fig.drawBoxes(names, ages, experiences)
        .fillSolid()
        .fillColor("pink")
        .fillIntensity(0.5)
        .borderShow()
        .labelNone();

    // Adjust the relative width of the boxes
    fig.boxWidthRelative(0.75);

    fig.autoclean(false);

    // Show the figure in a pop-up window
    fig.show();

    // Save the figure to a PDF file
    fig.save("example-boxes-ticklabels.pdf");
}

    Integration with CMake-based projects¶

    Integrating sciplot in a CMake-based project is very simple as shown next.

    Let's assume our CMake-based project consists of two files: main.cpp and CMakeLists.txt, whose contents are shown below:

    main.cpp 

    #include <sciplot/sciplot.hpp>
using namespace sciplot;

int main(int argc, char** argv)
{
    Vec x = linspace(0.0, PI, 200);

    Figure fig;

    fig.xlabel("x");
    fig.ylabel("y");

    fig.xrange(0.0, PI);
    fig.yrange(0.0, 6.0);

    fig.drawCurve(x, std::sin(x)).label("sin(x)");

    fig.save("my-sine-plot.pdf");
}

    CMakeLists.txt

    cmake_minimum_required(VERSION 3.0)

project(app)

find_package(sciplot)

add_executable(app main.cpp)

target_link_libraries(app sciplot::sciplot)

    In the CMakeLists.txt file, note the use of the command:

    find_package(sciplot)

    to find the header files of the sciplot library, and the command:

    target_link_libraries(app sciplot::sciplot)
    to link the executable target app against the sciplot library (sciplot::sciplot) using CMake's modern target-based design.

    To build the application, do:

    mkdir build && cd build
cmake .. -DCMAKE_PREFIX_PATH=/path/to/sciplot/install/dir
make

    Attention

    If sciplot has been installed system-wide, then the CMake argument CMAKE_PREFIX_PATH should not be needed. Otherwise, you will need to specify where sciplot is installed in your machine. For example:

    cmake .. -DCMAKE_PREFIX_PATH=$HOME/local

    assuming directory $HOME/local is where sciplot was installed to, which should then contain the following directory:

    $HOME/local/include/sciplot/

    where the sciplot header files are located.

    To execute the application, do:

    ./app
